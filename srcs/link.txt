KICKSTART_ADDR = 0x0f80000;      /* Kickstart address */
KICKSTART_SIZE = 0x0080000;      /* 512kB */
MEMORY
{
  rom (rx) : ORIGIN = KICKSTART_ADDR, LENGTH = KICKSTART_SIZE
}
SECTIONS
{
  .text :
  {
    *(startup)
    *(initstartup)
    *(CODE)
    *(generic)
    *(mainmenu)
    *(audiotest)
    *(memtest)
    *(irqcia)
    *(gfxtest)
    *(porttest)
    *(disktest)
    *(othertest)
    *(keyboardtest)
    *(autoconfig)
    *(terriblefire)
    *(.text)
    *(module)
    *(gfx)
    *(data)
    *(.data)
    *(.copperlist)
    *(.romEcsLowResCopper)
    *(.romEcsHiResCopper)
    *(.romAgaLowResCopper)
    *(.romAgaHiResCopper)
    *(.checksums)
    *(EXCLUDE_FILE(*autovec.o) *)   /* absorb any remaining orphan sections */
    . = KICKSTART_SIZE - 16;       /* autovec is 16 bytes, ends exactly at 512KB */
    *(.autovec)
  } > rom

  /DISCARD/ : { *(*) }             /* discard anything not explicitly placed */
}
